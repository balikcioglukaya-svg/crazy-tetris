    <script>
        // --- AYARLAR ---
        const canvas = document.getElementById('tetris');
        const context = canvas.getContext('2d');
        const nextCanvas = document.getElementById('next-canvas');
        const nextContext = nextCanvas.getContext('2d');
        
        context.scale(20, 20);
        nextContext.scale(15, 15); // Önizleme biraz daha küçük ölçekte (60px kutu için)

        const arena = createMatrix(12, 20);
        const player = { pos: {x: 0, y: 0}, matrix: null, score: 0, next: null };

        let dropCounter = 0;
        let dropInterval = 1000;
        let lastTime = 0;
        let isRunning = false;
        let isPaused = false;
        let animationId = null;

        // --- FONKSİYONLAR ---
        function createMatrix(w, h) {
            const matrix = [];
            while (h--) matrix.push(new Array(w).fill(0));
            return matrix;
        }

        const pieces = 'ILJOTSZ';
        function createPiece(type) {
            if (type === 'I') return [[0, 1, 0, 0], [0, 1, 0, 0], [0, 1, 0, 0], [0, 1, 0, 0]];
            if (type === 'L') return [[0, 2, 0], [0, 2, 0], [0, 2, 2]];
            if (type === 'J') return [[0, 3, 0], [0, 3, 0], [3, 3, 0]];
            if (type === 'O') return [[4, 4], [4, 4]];
            if (type === 'Z') return [[5, 5, 0], [0, 5, 5], [0, 0, 0]];
            if (type === 'S') return [[0, 6, 6], [6, 6, 0], [0, 0, 0]];
            if (type === 'T') return [[0, 7, 0], [7, 7, 7], [0, 0, 0]];
        }

        const colors = [null, '#FF0D72', '#0DC2FF', '#0DFF72', '#F538FF', '#FF8E0D', '#FFE138', '#3877FF'];

        function draw() {
            context.fillStyle = '#000';
            context.fillRect(0, 0, canvas.width, canvas.height);
            drawMatrix(arena, {x: 0, y: 0}, context);
            drawMatrix(player.matrix, player.pos, context);
        }

        function drawPreview() {
            nextContext.fillStyle = '#000';
            nextContext.fillRect(0, 0, nextCanvas.width, nextCanvas.height);
            if (player.next) {
                // 4x4'lük grid içinde ortala
                const offset = {
                    x: (4 - player.next[0].length) / 2,
                    y: (4 - player.next.length) / 2
                };
                drawMatrix(player.next, offset, nextContext);
            }
        }

        function drawMatrix(matrix, offset, ctx) {
            matrix.forEach((row, y) => {
                row.forEach((value, x) => {
                    if (value !== 0) {
                        ctx.fillStyle = colors[value];
                        ctx.fillRect(x + offset.x, y + offset.y, 1, 1);
                        ctx.lineWidth = 0.05;
                        ctx.strokeStyle = 'white';
                        ctx.strokeRect(x + offset.x, y + offset.y, 1, 1);
                    }
                });
            });
        }

        function merge(arena, player) {
            player.matrix.forEach((row, y) => {
                row.forEach((value, x) => {
                    if (value !== 0) {
                        arena[y + player.pos.y][x + player.pos.x] = value;
                    }
                });
            });
        }

        function rotate(matrix, dir) {
            for (let y = 0; y < matrix.length; ++y) {
                for (let x = 0; x < y; ++x) {
                    [matrix[x][y], matrix[y][x]] = [matrix[y][x], matrix[x][y]];
                }
            }
            if (dir > 0) matrix.forEach(row => row.reverse());
            else matrix.reverse();
        }

        function collide(arena, player) {
            const m = player.matrix;
            const o = player.pos;
            for (let y = 0; y < m.length; ++y) {
                for (let x = 0; x < m[y].length; ++x) {
                    if (m[y][x] !== 0 && (arena[y + o.y] && arena[y + o.y][x + o.x]) !== 0) {
                        return true;
                    }
                }
            }
            return false;
        }

        function arenaSweep() {
            let rowCount = 1;
            outer: for (let y = arena.length -1; y > 0; --y) {
                for (let x = 0; x < arena[y].length; ++x) {
                    if (arena[y][x] === 0) continue outer;
                }
                const row = arena.splice(y, 1)[0].fill(0);
                arena.unshift(row);
                ++y;
                player.score += rowCount * 10;
                rowCount *= 2;
            }
            updateScore();
        }

        function playerReset() {
            if (player.next === null) player.next = createPiece(pieces[pieces.length * Math.random() | 0]);
            player.matrix = player.next;
            player.next = createPiece(pieces[pieces.length * Math.random() | 0]);
            drawPreview();
            
            player.pos.y = 0;
            player.pos.x = (arena[0].length / 2 | 0) - (player.matrix[0].length / 2 | 0);
            
            if (collide(arena, player)) {
                gameOver();
            }
        }

        function playerDrop() {
            player.pos.y++;
            if (collide(arena, player)) {
                player.pos.y--;
                merge(arena, player);
                playerReset();
                arenaSweep();
            }
            dropCounter = 0;
        }

        function playerMove(dir) {
            player.pos.x += dir;
            if (collide(arena, player)) player.pos.x -= dir;
        }

        function playerRotate(dir) {
            const pos = player.pos.x;
            let offset = 1;
            rotate(player.matrix, dir);
            while (collide(arena, player)) {
                player.pos.x += offset;
                offset = -(offset + (offset > 0 ? 1 : -1));
                if (offset > player.matrix[0].length) {
                    rotate(player.matrix, -dir);
                    player.pos.x = pos;
                    return;
                }
            }
        }

        function updateScore() {
            document.getElementById('score-val').innerText = player.score;
            const level = Math.floor(player.score / 50) + 1;
            document.getElementById('level-val').innerText = level;
            dropInterval = Math.max(100, 1000 - (level - 1) * 100);
        }

        // --- NAVIGASYON & EKRAN YÖNETİMİ ---

        function showMainMenu() {
            isRunning = false;
            document.getElementById('menu-main').classList.remove('hidden');
            document.getElementById('menu-scores').classList.add('hidden');
            document.getElementById('game-container').classList.add('hidden');
            document.getElementById('top-bar').style.display = 'none';
        }

        function showHighScores() {
            updateHighScoreList();
            document.getElementById('menu-main').classList.add('hidden');
            document.getElementById('menu-scores').classList.remove('hidden');
            // Oyun game over ekranından buraya gelindiyse oyun konteynerini gizle
            document.getElementById('game-container').classList.add('hidden');
            document.getElementById('top-bar').style.display = 'none';
        }

        function quitToMenu() {
            showMainMenu();
        }

        function startGame() {
            // Ekranları ayarla
            document.getElementById('menu-main').classList.add('hidden');
            document.getElementById('menu-scores').classList.add('hidden');
            document.getElementById('game-container').classList.remove('hidden');
            document.getElementById('top-bar').style.display = 'flex'; // Üst barı aç

            // Overlayleri temizle
            document.getElementById('overlay-gameover').classList.add('hidden');
            document.getElementById('overlay-pause').classList.add('hidden');
            
            // Oyunu sıfırla
            arena.forEach(row => row.fill(0));
            player.score = 0;
            player.next = null;
            updateScore();
            
            // Geri sayım başlat
            startCountdown();
        }

        function startCountdown() {
            document.getElementById('overlay-countdown').classList.remove('hidden');
            let count = 3;
            const el = document.getElementById('count-num');
            el.innerText = count;
            
            const timer = setInterval(() => {
                count--;
                if (count > 0) {
                    el.innerText = count;
                } else {
                    clearInterval(timer);
                    document.getElementById('overlay-countdown').classList.add('hidden');
                    // Oyun başlasın
                    isRunning = true;
                    isPaused = false;
                    playerReset();
                    lastTime = performance.now();
                    dropCounter = 0;
                    update();
                }
            }, 1000);
        }

        function update(time = 0) {
            if (!isRunning || isPaused) return;
            const deltaTime = time - lastTime;
            lastTime = time;
            dropCounter += deltaTime;
            
            if (dropCounter > 2000) dropCounter = 0;
            if (dropCounter > dropInterval) playerDrop();
            
            draw();
            animationId = requestAnimationFrame(update);
        }

        function togglePause() {
            if (!isRunning) return;
            isPaused = !isPaused;
            if (isPaused) {
                document.getElementById('overlay-pause').classList.remove('hidden');
            } else {
                document.getElementById('overlay-pause').classList.add('hidden');
                lastTime = performance.now();
                update();
            }
        }

        function gameOver() {
            isRunning = false;
            document.getElementById('final-score').innerText = player.score;
            saveHighScore(player.score);
            document.getElementById('overlay-gameover').classList.remove('hidden');
        }

        // --- SKOR KAYIT (LocalStorage) ---
        function saveHighScore(score) {
            let scores = JSON.parse(localStorage.getItem('tetris_mobile_scores')) || [];
            scores.push(score);
            scores.sort((a, b) => b - a);
            scores = scores.slice(0, 5); // İlk 5'i tut
            localStorage.setItem('tetris_mobile_scores', JSON.stringify(scores));
        }

        function updateHighScoreList() {
            let scores = JSON.parse(localStorage.getItem('tetris_mobile_scores')) || [];
            const listEl = document.getElementById('hs-list-display');
            if (scores.length === 0) {
                listEl.innerHTML = "<li>Henüz skor yok</li>";
            } else {
                listEl.innerHTML = scores.map((s, i) => `<li>${i+1}. ${s}</li>`).join('');
            }
        }

        // --- KONTROLLER ---
        
        // Dokunmatik Buton Bağlantıları
        function bindTouch(id, action) {
            const btn = document.getElementById(id);
            const handler = (e) => {
                e.preventDefault(); // Zoom vs engelle
                if (isRunning && !isPaused) action();
            };
            btn.addEventListener('touchstart', handler, {passive: false});
            btn.addEventListener('mousedown', handler);
        }

        bindTouch('btn-left', () => playerMove(-1));
        bindTouch('btn-right', () => playerMove(1));
        bindTouch('btn-down', () => playerDrop());
        bindTouch('btn-rotate', () => playerRotate(1));

        // Klavye Desteği (PC Test için)
        document.addEventListener('keydown', event => {
            // Sadece oyun ekranındaysa çalışsın
            if(document.getElementById('game-container').classList.contains('hidden')) return;

            if (event.keyCode === 27) togglePause();
            if (!isRunning || isPaused) return;

            if (event.keyCode === 37) playerMove(-1);
            else if (event.keyCode === 39) playerMove(1);
            else if (event.keyCode === 40) playerDrop();
            else if (event.keyCode === 38 || event.keyCode === 81) playerRotate(1);
        });

        // Başlangıçta Menüyü Göster
        showMainMenu();

    </script>
